.swiper-wrapper {
    transition-timing-function: linear!important;
}
Js代码
var mySwiper = new Swiper ('.swiper', {
    loop: true,
    slidesPerView : 5,
    speed: 6000,//匀速时间
    autoplay: {
        delay: 0,
        stopOnLastSlide: false,
        disableOnInteraction: false,
    },
});
